    """视频指纹计算和匹配服务"""

        """使用 ffmpeg 提取视频帧"""
        # ffmpeg -i video.mp4 -vf "fps=1/2" frame_%04d.png
    
    def compute_phash(self, image_path: str) -> int:
        """计算感知哈希 (perceptual hash)"""
        # 1. 缩放至 8x8
        # 2. 灰度化
        # 3. DCT变换
        # 4. 取低频系数，中值为0，生成64位hash
    
    def compute_dhash(self, image_path: str) -> int:
        """计算差异哈希 (difference hash)"""
        # 1. 缩放至 9x8
        # 2. 灰度化
        # 3. 比较相邻像素亮度差异
    
    def hamming_distance(self, hash1: int, hash2: int) -> int:
        """计算两个hash的汉明距离"""
        return bin(hash1 ^ hash2).count('1')
    
    def scan_for_infringement(self, video_path: str, threshold: int = 10) -> List[MatchResult]:
