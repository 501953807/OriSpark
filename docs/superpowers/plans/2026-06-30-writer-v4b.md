**核心模型:**

```python
# Book (作品集合)
class Book(Base):
    __tablename__ = "books"
    id = Column(String, primary_key=True)
    creator_id = Column(String, nullable=False)
    title = Column(String(500), nullable=False)
    subtitle = Column(String(500), nullable=True)
    description = Column(Text, nullable=True)
    genre = Column(String(100))  # fiction/non-fiction/poetry/script/essay
    word_count = Column(Integer, default=0)
    cover_path = Column(String(500), nullable=True)
    status = Column(String(20), default="draft")  # draft|in_progress|completed|published
    license_type = Column(String(50), default="all_rights_reserved")
    created_at = Column(DateTime, default=current_timestamp)
    updated_at = Column(DateTime, default=current_timestamp, onupdate=current_timestamp)

# Chapter (章节)
class Chapter(Base):
    __tablename__ = "chapters"
    id = Column(String, primary_key=True)
    book_id = Column(String, ForeignKey("books.id"), nullable=False)
    title = Column(String(500), nullable=False)
    order_index = Column(Integer, nullable=False)
    word_count = Column(Integer, default=0)
    status = Column(String(20), default="draft")  # draft|review|final
    content_hash = Column(String(64), nullable=True)  # SHA-256 of content for plagiarism check
    created_at = Column(DateTime, default=current_timestamp)
    updated_at = Column(DateTime, default=current_timestamp, onupdate=current_timestamp)

# Scene (场景 - 章节内细分)
class Scene(Base):
    __tablename__ = "scenes"
    id = Column(String, primary_key=True)
    chapter_id = Column(String, ForeignKey("chapters.id"), nullable=False)
    title = Column(String(500), nullable=True)
    order_index = Column(Integer, nullable=False)
    word_count = Column(Integer, default=0)
    notes = Column(Text, nullable=True)  # 作者笔记
    created_at = Column(DateTime, default=current_timestamp)

# Plagiarism Check Result
class PlagiarismCheck(Base):
    __tablename__ = "plagiarism_checks"
    id = Column(String, primary_key=True)
    chapter_id = Column(String, nullable=True)  # null = book-level check
    content_hash = Column(String(64), nullable=False)
    similarity_score = Column(Numeric(5, 2), nullable=False)  # 0-100%
    matched_sources = Column(JSON, nullable=True)  # [{"url": "...", "similarity": 15.5, "matched_text": "..."}]
    checker = Column(String(50), default="internal")  # internal|turnitin|grammarly
    status = Column(String(20), default="completed")  # pending|completed|failed
    created_at = Column(DateTime, default=current_timestamp)

# EPUB Export History
class EPUBExport(Base):
    __tablename__ = "epub_exports"
    id = Column(String, primary_key=True)
    book_id = Column(String, nullable=False)
    file_path = Column(String(500), nullable=False)
    file_size = Column(Integer, nullable=True)
    chapter_count = Column(Integer, nullable=True)
    word_count = Column(Integer, nullable=True)
    created_at = Column(DateTime, default=current_timestamp)
```
